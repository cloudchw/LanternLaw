```typescript
// 将跨天范围拆分为两个不跨天的区间
const normalizeRange = (start, end) => {
  if (end >= start) {
    return [[start, end]];  // 不跨天
  }
  return [[start, 24*60], [0, end]];  // 跨天拆分
};

// 检查所有区间的组合是否重叠
```
