### 2.2 TimeRangeManager.ets (D:\GitHub\LanternLaw\entry\src\main\ets\models\TimeRange.ets)

**重叠检测算法** (第25-70行):
```typescript
static isOverlapping(range1: TimeRange, range2: TimeRange): boolean {
  const start1 = TimeRangeManager.timeToMinutes(range1.startTime);
  const end1 = TimeRangeManager.timeToMinutes(range1.endTime);
  const start2 = TimeRangeManager.timeToMinutes(range2.startTime);
  const end2 = TimeRangeManager.timeToMinutes(range2.endTime);

  // 处理跨天情况
  const range1SpansMidnight = end1 < start1;
  const range2SpansMidnight = end2 < start2;

  // 如果只有一个跨天，肯定不会重叠
  if (range1SpansMidnight !== range2SpansMidnight) {
    return false;
  }

  // 如果两个都跨天，检查是否真的重叠
  if (range1SpansMidnight && range2SpansMidnight) {
    // 两者都跨天，认为重叠（简化处理）
    return true;
  }

  // 两个都不跨天，正常比较
  const overlaps = !(end1 <= start2 || end2 <= start1);
  return overlaps;
}
```

**时间转换功能**:
```typescript
// 时间字符串转换为分钟数
static timeToMinutes(timeStr: string): number {
  const parts = timeStr.split(':');
  const hours = parseInt(parts[0], 10);
  const minutes = parseInt(parts[1], 10);
  return hours * 60 + minutes;
}

// 分钟数转换为时间字符串
static minutesToTime(minutes: number): string {
  const hours = Math.floor(minutes / 60) % 24;
  const mins = minutes % 60;
  return `${hours.toString().padStart(2, '0')}:${mins.toString().padStart(2, '0')}`;
}
```

**计算逻辑** (第84-99行):
```typescript
private calculateAddingDuration(): string {
  if (!this.addingStartTime || !this.addingEndTime) {
    return '0.00h';
  }

  const startMinutes = TimeRangeManager.timeToMinutes(this.addingStartTime);
  const endMinutes = TimeRangeManager.timeToMinutes(this.addingEndTime);

  let adjustedEndMinutes = endMinutes;
  if (endMinutes < startMinutes) {
    adjustedEndMinutes += 24 * 60; // 跨天处理
  }

  const hours = (adjustedEndMinutes - startMinutes) / 60;
  return hours.toFixed(2) + 'h';
}
```
