#### 2. 跨天重叠检测
```typescript
// TimeRange.ets - isOverlapping()
// 将跨天范围拆分为两个不跨天的区间
const normalizeRange = (start, end) => {
  if (end >= start) {
    return [[start, end]];           // 不跨天
  }
  return [[start, 24*60], [0, end]]; // 跨天拆分
};

// 检查所有区间的组合是否重叠
for (let i = 0; i < intervals1.length; i++) {
  for (let j = 0; j < intervals2.length; j++) {
    const s1 = intervals1[i][0], e1 = intervals1[i][1];
    const s2 = intervals2[j][0], e2 = intervals2[j][1];
    if (!(e1 <= s2 || e2 <= s1)) {
      return true; // 发现重叠
    }
  }
}
```
